
local BaseObject = class("BaseObject")
GUI.BaseObject = BaseObject

BaseObject.pos = {x=0, y=0}
BaseObject.globalPos = {x=0, y=0}
BaseObject.size = {x=0, y=0}
--BaseObject.color = GUI.theme.PRIMARY
--BaseObject.outlineColor = GUI.theme.OUTLINE_PRIMARY
--BaseObject.textColor = GUI.theme.TEXT_PRIMARY
BaseObject.text = "Def"

BaseObject.children = {}
BaseObject.parent = nil
BaseObject.parenter = nil

BaseObject.renderingEnabled = true

BaseObject.childrenPosOffset = {x=0, y=0}

BaseObject.childrenRenderingEnabled = true

BaseObject.scriptsEnabled = true
BaseObject.scripts = {}

--BaseObject.renderLayer = nil

function BaseObject:render(data) end


function BaseObject:initBaseObject(type, parent)
    self.type = type
    self.parenter = self
    self:setParent((parent or GUI.root).parenter)
end

function BaseObject:renderBaseObject()
    if self.renderingEnabled then
        if self.scriptsEnabled then
            for _, script in pairs(self.scripts) do
                script(self)
            end
        end
         
        self:render()
        
        if self.childrenRenderingEnabled then
            for _, child in pairs(self.children) do
                child:renderBaseObject()
            end
        end
    end
end



function BaseObject:setParent(obj)
    self.parenter.parent = obj.parenter
    if not table.hasValue(obj.children, self.parenter) then
        obj.children[#obj.children + 1] = self.parenter
    end
end
function BaseObject:unparent()
    if self.parenter.parent == GUI.root then
        return
    end
    table.removeByValue(self.parenter.parent.children, self.parenter)
    self.parenter.parent = GUI.root
end

function BaseObject:setPos(x, y)
    self.pos.x, self.pos.y = x, y
end
function BaseObject:setGlobalPos(x, y)
    local parent = self.parent
    
    if parent == GUI.root then
        x = x - parent.pos.x - parent.childrenPosOffset.x
        y = y - parent.pos.y - parent.childrenPosOffset.y
    end
    
    while parent ~= GUI.root do
        x = x - parent.pos.x - parent.childrenPosOffset.x
        y = y - parent.pos.y - parent.childrenPosOffset.y
        parent = parent.parent
    end
    
    self.pos.x, self.pos.y = x, y
end
function BaseObject:getPos()
    return self.pos
end
function BaseObject:getGlobalPos()
    local x, y = self.pos.x + self.childrenPosOffset.x, self.pos.y + self.childrenPosOffset.y
    local parent = self.parent

    if parent == GUI.root then
        x = x + parent.pos.x + parent.childrenPosOffset.x
        y = y + parent.pos.y + parent.childrenPosOffset.y
    end

    while parent ~= GUI.root do
        x = x + parent.pos.x + parent.childrenPosOffset.x
        y = y + parent.pos.y + parent.childrenPosOffset.y
        parent = parent.parent
    end

    self.globalPos.x = x
    self.globalPos.y = y

    return self.globalPos
end

function BaseObject:setSize(x, y)
    self.size.x, self.size.y = x, y
end
function BaseObject:getSize()
    return self.size
end

function BaseObject:enableChildren(bool)
    self.childrenRenderingEnabled = bool
end


function BaseObject:topOrder()
    local children = self.parent.children
    local selfIndex = table.keyFromValue(children, self)
    local lastIndex = #children

    while selfIndex ~= lastIndex do
        children[selfIndex], children[selfIndex + 1] = children[selfIndex + 1], children[selfIndex]

        selfIndex = table.keyFromValue(children, self)
        lastIndex = #children
    end
end
function BaseObject:downOrder()
    local children = self.parent.children
    local selfIndex = table.keyFromValue(children, self)
    children[selfIndex], children[1] = children[1], children[selfIndex]
end
function BaseObject:onTop()
    return #self.parent.children == table.keyFromValue(self.parent.children, self)
end

end
