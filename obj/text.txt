local Text = class("Text", GUI.BaseObject)
GUI.Text = Text

local textColor = GUI.theme.TEXT_PRIMARY

function Text:initialize(parent)
    self:initBaseObject("Text", parent)

    self:setText("Text")
end

function Text:render()
    local pos = self:getGlobalPos()
    local size = self.size
    render.setRGBA(textColor.r, textColor.g, textColor.b, textColor.a)
    render.drawText(pos.x, pos.y, self.text)
end

function Text:setText(str)
    self.text = str
    local xs, ys = GUI.getTextRectSize(str)
    self.size.x = xs + 4
    self.size.y = ys * #string.explode("\n", str) + 4
end
