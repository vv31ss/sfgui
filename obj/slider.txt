local Slider = class("Slider", GUI.BaseObject)
GUI.Slider = Slider

local handleColor = GUI.theme.ACCENT
local color = GUI.theme.PRIMARY
local textColor = GUI.theme.TEXT_PRIMARY
local outlineColor = GUI.theme.OUTLINE_PRIMARY

function Slider:initialize(parent)
    self:initBaseObject("Slider", parent)
    
    self.pressing = false
    
    self.handleOffset = 0
    
    self.oldValue = 0
    self.value = 0
    
    self.handleSizeX = 0
    self.handleSizeY = 0
    self.handleSize = 0
    self:setHandleSize(24)
    
    self:setSize(96, 14)
    
    
    self.fillEnabled = true
    
    
    self.pressedCallback = function(value) end
    self.releasedCallback = function(value) end
    self.holdingCallback = function(value) end
    
    self.valueChangedCallback = function(value) end
    self.valueChangedOnceCallback = function(value) end
end
function Slider:render()
    local pos = self:getGlobalPos()
    local size = self.size
    

    render.setRGBA(color.r, color.g, color.b, color.a)
    render.drawRect(pos.x, pos.y, size.x, size.y)
    
    
    local vertical = size.x < size.y
    
    if self.fillEnabled then
        render.setRGBA(255 - handleColor.r, 255 - handleColor.g, 255 - handleColor.b, handleColor.a / 1.5)
        if vertical then
            render.drawRect(pos.x, pos.y, self.handleSizeX, self.handleSizeY + self.value * (size.y - self.handleSizeY))
        else
            render.drawRect(pos.x, pos.y, self.handleSizeX + self.value * (size.x - self.handleSizeX), self.handleSizeY)
        end
    end
    
    render.setRGBA(handleColor.r, handleColor.g, handleColor.b, handleColor.a)
    if vertical then
        render.drawRect(pos.x, pos.y + self.value * (size.y - self.handleSizeY), self.handleSizeX, self.handleSizeY)
    else
        render.drawRect(pos.x + self.value * (size.x - self.handleSizeX), pos.y, self.handleSizeX, self.handleSizeY)
    end
    
    
    
    render.setRGBA(outlineColor.r, outlineColor.g, outlineColor.b, outlineColor.a)
    render.drawRectOutline(pos.x, pos.y, size.x, size.y)


    
    GUI.doReverseCall(function()
        local cursor = GUI.cursorPos
        
        if self.oldValue ~= self.value then
            self.valueChangedCallback(self.value)
            self.oldValue = self.value
        end
        
        if cursor.x > pos.x and cursor.y > pos.y and cursor.x < pos.x + size.x and cursor.y < pos.y + size.y then
            local hposx, hposy    
            if vertical then
                hposx = pos.x
                hposy = pos.y + self.value * (size.y - self.handleSizeY)
            else
                hposx = pos.x + self.value * (size.x - self.handleSizeX)
                hposy = pos.y
            end
            
            if cursor.x > hposx and cursor.y > hposy and cursor.x < hposx + self.handleSizeX and cursor.y < hposy + self.handleSizeY then
                render.setRGBA(255 - handleColor.r, 255 - handleColor.g, 255 - handleColor.b, handleColor.a / 3)
                render.drawRect(hposx, hposy, self.handleSizeX, self.handleSizeY)
                if not self.pressing and GUI.mouseLeftPressed then
                    if vertical then
                        self.handleOffset = hposy - (cursor.y - self.handleSizeY / 2)
                    else
                        self.handleOffset = hposx - (cursor.x - self.handleSizeX / 2)
                    end
                end
            end
            
            if GUI.mouseLeftPressed then
                if not self.pressing then
                    self.pressedCallback(self.value)
                end
                self.pressing = true
                GUI.interruptReverse = true
            end
            if GUI.mouseLeftReleased and self.pressing then
                self.valueChangedOnceCallback(self.value)
                self.releasedCallback(self.value)
                self.pressing = false
                self.handleOffset = 0
            end
        elseif self.pressing and not GUI.mouseLeftHolding then
            self.valueChangedOnceCallback(self.value)
            self.releasedCallback(self.value)
            self.pressing = false
            self.handleOffset = 0
        end
        if self.pressing then
            self.holdingCallback(self.value)
            if vertical then
                self.value = math.clamp(math.remap(
                    cursor.y + self.handleOffset,
                    pos.y + self.handleSizeY / 2,
                    pos.y + size.y - self.handleSizeY / 2,
                    0, 1), 0, 1)
            else
                self.value = math.clamp(math.remap(
                    cursor.x + self.handleOffset,
                    pos.x + self.handleSizeX / 2,
                    pos.x + size.x - self.handleSizeX / 2,
                    0, 1), 0, 1)
            end
        end
    end)
end

function Slider:setHandleSize(num)
    local vertical = self.size.x < self.size.y
    if vertical then
        self.handleSizeX = self.size.x
        self.handleSizeY = num
    else
        self.handleSizeX = num
        self.handleSizeY = self.size.y
    end
    self.handleSize = num
end

function Slider:setSize(x, y)
    self.size.x, self.size.y = x, y
    self:setHandleSize(self.handleSize)
end

function Slider:setPressedCallback(func)
    self.pressedCallback = func
end
function Slider:setReleasedCallback(func)
    self.releasedCallback = func
end
function Slider:setHoldingCallback(func)
    self.holdingCallback = func
end

function Slider:setValueChangedCallback(func)
    self.valueChangedCallback = func
end

function Slider:setValueChangedOnceCallback(func)
    self.valueChangedOnceCallback = func
end

function Slider:setValue(num)
    self.value = num
    if self.oldValue ~= self.value then
        self.valueChangedOnceCallback(self.value)
    end
end

function Slider:getValue()
    return self.value
end

function Slider:enableFill(bool)
    self.fillEnabled = bool
end
