local Tab = class("Tab", GUI.BaseObject)
GUI.Tab = Tab

local color = GUI.theme.PRIMARY
local textColor = GUI.theme.TEXT_PRIMARY
local outlineColor = GUI.theme.OUTLINE_PRIMARY

function Tab:initialize(parent)
    self:initBaseObject("Tab", parent)

    self:setSize(40, 16)
    
    self.pressing = false
    
    self.text = "Tab"

    self.active = false
    
    if parent.type == "TabHolder" then
        for _, obj in pairs(parent.children) do
            if obj ~= self and obj.type == "Tab" and obj.active == true then
                break
            end
            if obj == self then
                self.active = true
                break
            end
        end
    end
end
function Tab:render()
    self:setChildrenPosOffset(0, 0)
    local pos = self:getGlobalPos()
    self:setChildrenPosOffset(-self.pos.x, -self.pos.y)

    local size = self.size

    if self.active then
        render.setRGBA((255 - color.r / 2), (255 - color.g / 2), (255 - color.b / 2), color.a)
    else
        render.setRGBA(color.r, color.g, color.b, color.a)
    end
    render.drawRect(pos.x, pos.y, size.x, size.y)

    if self.active then
        render.setRGBA(outlineColor.r, outlineColor.g, outlineColor.b, outlineColor.a)
        render.drawRectOutline(pos.x, pos.y, size.x, size.y)
    end

    if string.trim(self.text) ~= "" then
        if self.active then
            render.setRGBA((255 - textColor.r), (255 - textColor.g), (255 - textColor.b), textColor.a)
        else
            render.setRGBA(textColor.r, textColor.g, textColor.b, textColor.a)
        end
        render.drawText(pos.x + size.x / 2, pos.y, self.text, TEXT_ALIGN.CENTER)
    end
    
    self:enableChildren(self.active)

    GUI.doReverseCall(function()
        local cursor = GUI.cursorPos
        
        if cursor.x > pos.x and cursor.y > pos.y and cursor.x < pos.x + size.x and cursor.y < pos.y + size.y then
            if not self.active then
                render.setRGBA(255 - color.r, 255 - color.g, 255 - color.b, color.a / 4)
                render.drawRect(pos.x, pos.y, size.x, size.y)
            end
            if GUI.mouseLeftPressed then
                self.pressing = true
                GUI.interruptReverse = true
                if self.parent.type == "TabHolder" then
                    for _, tab in pairs(self.parent.children) do
                        if tab.type ~= "Tab" then
                            continue
                        end
                        tab.active = false
                    end
                end
                self.active = not self.active
            end
            if GUI.mouseLeftReleased and self.pressing then
                self.pressing = false
            end
        elseif self.pressing then
            self.pressing = false
        end
    end)
end
