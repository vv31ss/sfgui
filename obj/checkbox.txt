local CheckBox = class("CheckBox", GUI.BaseObject)
GUI.CheckBox = CheckBox

local color_true = GUI.theme.CHECKBOX_TRUE
local color_false = GUI.theme.CHECKBOX_FALSE
local outlineColor = GUI.theme.OUTLINE_PRIMARY
local textColor = GUI.theme.TEXT_PRIMARY

function CheckBox:initialize(parent)
    self:initBaseObject("CheckBox", parent)
    
    self:setSize(16)
    
    self.active = false
    self.oldActive = false
    
    self.text = "CheckBox"
    
    self.callback = function(bool) end
end
function CheckBox:render()
    local pos = self:getGlobalPos()
    local size = self.size
    
    local radius = size.x / 2
    
    if self.active then
        render.setRGBA(color_true.r, color_true.g, color_true.b, color_true.a)
    else
        render.setRGBA(color_false.r, color_false.g, color_false.b, color_false.a)
    end
    render.drawFilledCircle(pos.x + radius, pos.y + radius, radius - 1)

    render.setRGBA(outlineColor.r, outlineColor.g, outlineColor.b, outlineColor.a)
    render.drawCircle(pos.x + radius, pos.y + radius, radius)
    
    if string.trim(self.text) ~= "" then
        render.setRGBA(textColor.r, textColor.g, textColor.b, textColor.a)
        render.drawText(pos.x + size.x + 4, pos.y + size.y / 2 - 8, self.text, TEXT_ALIGN.LEFT)
    end
    

    GUI.doReverseCall(function()
        local cursor = GUI.cursorPos

        if self.active ~= self.oldActive then
            self.oldActive = self.active
            self.callback(self.active)
        end
        
        if cursor.x > pos.x and cursor.y > pos.y and cursor.x < pos.x + size.x and cursor.y < pos.y + size.y then
            if self.active then
                render.setRGBA(255 - color_true.r, 255 - color_true.g, 255 - color_true.b, 55)
            else
                render.setRGBA(255 - color_false.r, 255 - color_false.g, 255 - color_false.b, 55)
            end
            render.drawFilledCircle(pos.x + radius, pos.y + radius, radius - 1)
            
            if GUI.mouseLeftPressed then
                GUI.interruptReverse = true
            end
            
            if GUI.mouseLeftReleased then
                self.active = not self.active
                self.oldActive = self.active
                self.callback(self.active)
            end
        end
    end)
end

function CheckBox:setSize(num)
    self.size.x = num
    self.size.y = num
end

function CheckBox:setCallback(func)
    self.callback = func
end
