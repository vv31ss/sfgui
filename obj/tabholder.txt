local TabHolder = class("TabHolder", GUI.BaseObject)
GUI.TabHolder = TabHolder

function TabHolder:initialize(parent)
    self:initBaseObject("TabHolder", parent)
    
    self.maxTabSize = 512
    
    self:setSize(128, 16)
end
function TabHolder:render()
    self:setChildrenPosOffset(0, 0)
    
    local pos = self:getGlobalPos()
    local size = self.size

    local tabsCount = 0
    for i, obj in pairs(self.children) do
        if obj.type == "Tab" then
            tabsCount = tabsCount + 1
        end
    end
    local div = math.min(math.floor(self.size.x / tabsCount), self.maxTabSize)
    local i = 0
    for _, obj in pairs(self.children) do
        if obj.type ~= "Tab" then
            continue
        end
        i = i + 1
        obj:setPos(div * (i - 1), self.pos.y)
        obj:setSize(div, self.size.y)
    end
    
    self:setChildrenPosOffset(-self.pos.x, -self.pos.y)
end

function TabHolder:setMaxTabSize(num)
    self.maxTabSize = num
end
